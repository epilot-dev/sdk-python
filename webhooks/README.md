    auth=shared.Auth(

            key_name="corrupti",

        oauth_config=shared.OAuthConfig(
            client_id="vel",

    ),
    creation_time="2021-04-27T12:01:13.000Z",

    filter=shared.Filter(
        key_to_filter="magnam",
        supported_values=[

            "suscipit",
        ],
    ),
    http_method="PATCH",
